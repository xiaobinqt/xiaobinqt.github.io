    package main

import (
	"fmt"
)

func isAnagram(s string, t string) bool {
	if len(s) != len(t) {
		return false
	}

	var (
		m1 = make(map[int32]int, 0)
		m2 = make(map[int32]int, 0)
	)

	for _, each := range s {
		if _, ok := m1[each]; ok {
			m1[each]++
		} else {
			m1[each] = 1
		}
	}

	for _, each := range t {
		if _, ok := m1[each]; ok {
			m2[each]++
		} else {
			m2[each] = 1
		}
	}

	fmt.Println(m1, m2)

	for key, count := range m1 {
		fmt.Println(count, key)
		if _, ok := m2[key]; !ok {
			return false
		} else {
			if m2[key] != count {
				return false
			}
		}
	}

	return true
}

func main() {
	fmt.Println(isAnagram("abc", "bac"))
}
