    package main

import (
	"fmt"
)

func reverseString(s []byte) {
	for left, right := 0, len(s)-1; left < right; left++ {
		s[left], s[right] = s[right], s[left]
		right--
	}
}

func main() {
	x := []byte{'1', '2', '3'}
	reverseString(x)
	fmt.Println(string(x))
}
