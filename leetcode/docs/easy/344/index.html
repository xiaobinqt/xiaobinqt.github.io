    package main

import (
	"fmt"
)

func reverseString(s []byte) {
	length := len(s)
	if length == 0 {
		return
	}

	left, right := 0, length-1

	for left < right {
		tmp := s[left]
		s[left] = s[right]
		s[right] = tmp
		left++
		right--
	}

	return
}

func main() {
	x := []byte{'1', '2', '3'}
	reverseString(x)
	fmt.Println(string(x))
}
