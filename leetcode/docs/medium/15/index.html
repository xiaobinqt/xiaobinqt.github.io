func threeSum(nums []int) [][]int {
	sort.Ints(nums)
	res := [][]int{}
	// æ‰¾å‡º a + b + c = 0

		n1 := nums[i]

		}

		// å»é‡ a
		if i > 0 && n1 == nums[i-1] {
			continue
		}

		left, right := i+1, len(nums)-1

		for left < right {
			n2, n3 := nums[left], nums[right]
			if n1+n2+n3 == 0 {
				res = append(res, []int{n1, n2, n3})

			}
		}
	}

	return res
}
