    package main

import "fmt"

type ListNode struct {
	Val  int
	Next *ListNode
}

func removeNthFromEnd(head *ListNode, n int) *ListNode {
	if head == nil {
		return head
	}

	length := 0
	tmpc := head
	for tmpc != nil {
		//fmt.Println(tmpc.Val)
		length++
		tmpc = tmpc.Next
	}

	//fmt.Println("é•¿åº¦ä¸º...", length)

	dummy := &ListNode{
		Val:  0,
		Next: head,
	}

	cur := dummy
	for i := 1; i < length-n+1; i++ {
		cur = cur.Next
	}
	cur.Next = cur.Next.Next
	return dummy.Next
}

func main() {
	ret := removeNthFromEnd(&ListNode{
		Val: 1,
		Next: &ListNode{
			Val: 2,
			Next: &ListNode{
				Val: 3,
				Next: &ListNode{
					Val: 4,
					Next: &ListNode{
						Val:  5,
						Next: nil,
					},
				},
			},
		},
	}, 2)
	fmt.Println(ret)
}
