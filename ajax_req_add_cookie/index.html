$.ajax({
    url: "/nodered/nodes",
    headers: {
        Accept: "text/html",
    },
    xhrFields: {
        withCredentials: true // 携带 cookie 信息
    },
    success: function (data) {
        console.log(data)
        $("#red-ui-palette-container").html(data)
    },
    error: function (jqXHR) {
        console.log(jqXHR)
    }
});
